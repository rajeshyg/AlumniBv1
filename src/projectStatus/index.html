async function loadProjectData() { try { // Use a relative path since both files are now in the same directory const response = await fetch('./project.json'); if (!response.ok) { throw new Error(`HTTP error! status: ${response.status}`); } const data = await response.json(); console.log('Loaded project data:', data); return data; } catch (error) { console.error('Error loading project data:', error); document.getElementById('project-status').innerHTML = `
Error loading project data: ${error.message}
`; throw error; } }
